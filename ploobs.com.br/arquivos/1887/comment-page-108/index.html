Phong Tessellation with Geometry Shader


First of all, implementing Phong Tessellation using Geometry Shader is NOT a good idea at all since now we have the Directx 11 Tessellator unit !!!!
But when i implemented this we did not. And after all, the DirectX 11 implementation uses almost the same shaders idea showed in this post.

We recieve a low poly model, send it to a geometry shader that expands the triangle count according to the Phong Tessellation method, then stream out the produced vertices. We can send these vertex again to the pipeline and expand more the geometry.
After some passes, we just pick the produced vertex buffer and send again to the pipeline, but now we just draw it on the screen.

The following shader show how to expand the vertices in the geometry shader

//--------------------------------------------------------------------------------------
// Constant Buffer Variables
//--------------------------------------------------------------------------------------
Texture2D txDiffuse : register( t0 );
SamplerState samLinear : register( s0 );

cbuffer cbNeverChanges : register( b0 )
{
    matrix View;
    matrix World;
};

cbuffer cbChangeOnResize : register( b1 )
{
    matrix Projection;
};

//--------------------------------------------------------------------------------------
struct VS_INPUT
{
    float4 Pos : POSITION;
    float2 Tex : TEXCOORD0;
    float3 Norm : NORMAL;
};

struct PS_INPUT
{
    float4 Pos : SV_POSITION;
    float2 Tex : TEXCOORD0;
    float3 Norm : TEXCOORD1;
};

//--------------------------------------------------------------------------------------
// Vertex Shader GS
//--------------------------------------------------------------------------------------
VS_INPUT VSGS( VS_INPUT input )
{
    VS_INPUT output = (VS_INPUT)0;
    output.Pos = input.Pos;
    output.Tex = input.Tex;
    output.Norm = input.Norm;
    return output;
}

//--------------------------------------------------------------------------------------
// Vertex Shader
//--------------------------------------------------------------------------------------
VS_INPUT VS( VS_INPUT input )
{
    VS_INPUT output = (VS_INPUT)0;
    output.Norm = mul( input.Norm, (float3x3)World );
    output.Pos  = mul( input.Pos,World );
	output.Pos  = mul( output.Pos  , View );
	output.Pos  = mul( output.Pos  , Projection );
	output.Tex  = input.Tex;
    return output;
}

///AUXs///

struct newPoint
{
float3 Pos;
float3 Norm;
float2 Tex;
};

float3 GetBaricentro(float u, float v, float3 pt1,float3 pt2,float3 pt3 )
{
  return pt1 * u + pt2 * v + (1-u-v) * pt3;
}

float2 GetBaricentro2(float u, float v, float2 pt1,float2 pt2,float2 pt3 )
{
  return pt1 * u + pt2 * v + (1-u-v) * pt3;
}

float3 OrthogonalProjectionOnPlane(float3 vetor,float3 local, float3 normal)
{
  return (vetor -  dot( vetor - local, normal) * normal);
}

newPoint GetPonto(float u, float v, float alfa, float3 pt1,float3 pt2,float3 pt3, float3 n1,float3 n2,float3 n3,float2 tex1, float2 tex2, float2 tex3)
{
	newPoint np;
	float w = 1 - u - v;
	float3 calc =  u*u * pt1 + v * v * pt2 + w * w * pt3 +
	 u*v* ( OrthogonalProjectionOnPlane(pt2,pt1,n1) + OrthogonalProjectionOnPlane(pt1,pt2,n2)  ) +
	 v*w * ( OrthogonalProjectionOnPlane(pt3,pt2,n2) + OrthogonalProjectionOnPlane(pt2,pt3,n3)  ) +
	 w*u * ( OrthogonalProjectionOnPlane(pt1,pt3,n3) + OrthogonalProjectionOnPlane(pt3,pt1,n1)  ) ;	 

	np.Pos = alfa * GetBaricentro(u,v,pt1,pt2,pt3) +  (1 - alfa) * calc;
	np.Norm = GetBaricentro(u,v,n1,n2,n3);
	np.Tex = GetBaricentro2(u,v,tex1,tex2,tex3);
	return np;	

}

void CreatePoint(newPoint ponto0,newPoint ponto1,newPoint ponto2,inout TriangleStream OutputStream)
{
	VS_INPUT output = (VS_INPUT)0;
	output.Norm = float4(ponto0.Norm,0);
	output.Pos = float4(ponto0.Pos,1);
	output.Tex = ponto0.Tex;
    OutputStream.Append( output );    	

    output = (VS_INPUT)0;
	output.Norm = float4(ponto1.Norm,0);
	output.Pos = float4(ponto1.Pos,1);
	output.Tex = ponto1.Tex;
    OutputStream.Append( output );    	

    output = (VS_INPUT)0;
	output.Norm = float4(ponto2.Norm,0);
	output.Pos = float4(ponto2.Pos,1);
	output.Tex = ponto2.Tex;
    OutputStream.Append( output );    	

    OutputStream.RestartStrip();
}

//--------------------------------------------------------------------------------------
[maxvertexcount(9)]
void GSTESS( triangle VS_INPUT input[3], inout TriangleStream OutputStream )
{

	newPoint resp0 = GetPonto(0, 1,alfa , input[0].Pos,input[1].Pos,input[2].Pos, input[0].Norm,input[1].Norm,input[2].Norm, input[0].Tex,input[1].Tex,input[2].Tex);
	newPoint resp1 = GetPonto(1, 0,alfa , input[0].Pos,input[1].Pos,input[2].Pos, input[0].Norm,input[1].Norm,input[2].Norm, input[0].Tex,input[1].Tex,input[2].Tex);

The Expand pass uses the commom vertex shader and the GSTESS Geometry shader.
The draw pass uses the commom vertex shader, the GEOMETRYSHADER (not necessary =P) and the PS pixel shader.
The results are not so good (to be good we need lots of expand passes … that generates looooooots of vertex data ….)
When i implemented this, i made the expand passes only once in the beginning, and after i just used the produced vertices.
