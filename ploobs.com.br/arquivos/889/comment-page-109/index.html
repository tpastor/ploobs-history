Se meu jogo me escutasse


Olá pessoal, a algum tempo atrás fiz um post sobre SpeechSynthesis e agora estou fazendo um para dar continuidade ao reconhecimento e fala.

O projeto foi desenvolvido em Xna 4.0 e tem por objetivo demonstrar como integrar XNA com o  SpeechRecognition de uma forma simples e prática.


Após criar um projeto simples em xna adicione o System.Speech às referências

Conceitos básicos

Gramática

Assim como na nossa língua, só reconhecemos e interpretamos palavras que são conhecidas e que já escutamos, temos de criar uma gramática de palavras que o nosso sistema interpretará

 private Grammar CreateCustomGrammar()
        {
            GrammarBuilder grammarBuilder = new GrammarBuilder();
            grammarBuilder.Append(new Choices("up", "right", "left", "down","shoot","hy"));
            return new Grammar(grammarBuilder);
        }

Neste exemplo reconheceremos: up,down,left,right e hy

Reconhecimento x Hipótese

O reconhecedor de voz “escuta” a palavra dita porém assim como nós fazemos ao não ouvir uma palavra não reconhecida, associamos a algo já conhecido que não necessáriamente será o que queríamos dizer. Por exemplo se eu falar “vai” será interpretado pela nossa gramática como “right”

Código

using System;
using System.Collections.Generic;
using System.Linq;
using Microsoft.Xna.Framework;
using Microsoft.Xna.Framework.Audio;
using Microsoft.Xna.Framework.Content;
using Microsoft.Xna.Framework.GamerServices;
using Microsoft.Xna.Framework.Graphics;
using Microsoft.Xna.Framework.Input;
using Microsoft.Xna.Framework.Media;
using System.Speech.Recognition;

namespace SpeechTest2
{

    public class Game1 : Microsoft.Xna.Framework.Game
    {

        Grammar grammar;
        SpeechRecognitionEngine recognizer = new SpeechRecognitionEngine(); 

        GraphicsDeviceManager graphics;
        SpriteBatch spriteBatch;
        Texture2D car;
        float angle;
        float speed;
        SpriteFont font;

        string heared, hipo;

        Vector2 position;
        Vector2 direction;

        ///
        /// Criando uma gramática
        ///
        ///
        private Grammar CreateCustomGrammar()
        {
            GrammarBuilder grammarBuilder = new GrammarBuilder();
            //Definindo as palavras reconhecidas
            grammarBuilder.Append(new Choices("up", "right", "left", "down","shoot","hy"));
            return new Grammar(grammarBuilder);
        }

        public void InitRecognizeEngine()
        {

            grammar = CreateCustomGrammar();

            // Apagando a gramática anterior para podermos usar somente a nossa
            recognizer.UnloadAllGrammars();

            //Carregando o nosso conjunto de palavras
            recognizer.LoadGrammar(grammar);

            //Definindo o metodo de reconhecimento
            recognizer.SpeechRecognized +=
       new EventHandler(recognizer_SpeechRecognized);

            //Definindo o método de hipóteses
            recognizer.SpeechHypothesized +=
            new EventHandler
                            (recognizer_SpeechHypothesized);
        }

        private void recognizer_SpeechRecognized(object sender,
                        SpeechRecognizedEventArgs e)
        {
            heared = e.Result.Text;

            switch (heared)
            {
                case "up":
                    angle = 0;
                    direction = new Vector2(0, -1);
                    break;
                case "down":
                    angle = MathHelper.ToRadians(180);
                    direction = new Vector2(0, 1);
                    break;
                case "right":
                    angle = MathHelper.ToRadians(90);
                    direction = new Vector2(1, 0);
                    break;
                case "left":
                    angle = MathHelper.ToRadians(-90);
                    direction = new Vector2(-1, 0);
                    break;

                default:
                    break;
            }

        }

        private void recognizer_SpeechHypothesized(object sender,
                        SpeechHypothesizedEventArgs e)
        {
            hipo = e.Result.Text;
        }

        public Game1()
        {

            speed = 1;
            angle = 0;
            graphics = new GraphicsDeviceManager(this);

            position = new Vector2(graphics.PreferredBackBufferWidth/2, graphics.PreferredBackBufferHeight/2);
            Content.RootDirectory = "Content";

            InitRecognizeEngine();

            //Setando para utilizar o microfone para reconhecimento
            recognizer.SetInputToDefaultAudioDevice();

            //Reconhecimento múltiplo significa que o reconhecedor continuará ativo
            // após o primeiro reconhecimento
            recognizer.RecognizeAsync(RecognizeMode.Multiple);

        }

        protected override void Initialize()
        {

            base.Initialize();
        }

        protected override void LoadContent()
        {

            spriteBatch = new SpriteBatch(GraphicsDevice);
            car = Content.Load("car");
            font = Content.Load("SpriteFont1");

        }

        protected override void Draw(GameTime gameTime)
        {
            GraphicsDevice.Clear(Color.CornflowerBlue);

            Rectangle ret =  new Rectangle(0, 0, car.Width, car.Height);
            Vector2 origin = new Vector2(ret.Width/2,ret.Height/2);
            spriteBatch.Begin();

            CalcPosition();

            spriteBatch.Draw(car, position,ret, Color.White,angle,origin,0.5f, SpriteEffects.None, 0);
            spriteBatch.DrawString(font, "Hipotese: " + hipo + "Usado: " + heared, Vector2.Zero, Color.Red);
            spriteBatch.End();

            base.Draw(gameTime);
        }

        public void CalcPosition()
        {

            position = position+direction*speed;

        }
    }
}
